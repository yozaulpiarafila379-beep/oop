Object Oriented Programming (OOP)

Object Oriented Programming (OOP) adalah paradigma pemrograman yang berfokus pada objek sebagai unit utama. Setiap objek memiliki state (data) dan behavior (perilaku). Tujuan OOP adalah membuat kode yang modular, reusable, dan mudah dipelihara dalam sistem yang kompleks.

1. Class & Object (Car)

Contoh pendefinisian class Car dengan dua attribute (model dan color) serta dua method (start dan stop).

Contoh Kode 1: Class Car dengan Metode Start & Stop

class Car {
  String? model;
  String? color;

  void start() {
    print('Car started!');
  }

  void stop() {
    print('Car stopped!');
  }
}

void main() {
  Car myCar = Car();
  myCar.model = "Sedan";
  myCar.color = "Merah";

  myCar.start();
  myCar.stop();
}

Contoh Kode 2: Class Car dengan Menampilkan Properti

class Car {
  String? model;
  String? color;
}

void main() {
  Car myCar = Car();
  myCar.model = "Sedan";
  myCar.color = "Merah";

  print(myCar.model);
  print(myCar.color);
}

Contoh Kode 3: Class Car dengan Konstruktor Posisional

class Car {
  String model;
  String color;

  Car(this.model, this.color);

  void start() {
    print('Engine Online: $model');
  }
}

void main() {
  Car myCar = Car("Sedan", "Merah");

  myCar.start();
  print(myCar.color);
}

2. Contoh Kelas Lain dalam OOP

Contoh Kode 4: Class Mahasiswa

class Mahasiswa {
  String nama;
  String nim;
  double ipk;

  Mahasiswa(this.nama, this.nim, this.ipk);

  void tampilkanData() {
    print('Nama : $nama');
    print('NIM : $nim');
    print('IPK : $ipk');
  }
}

void main() {
  Mahasiswa mhs1 = Mahasiswa("Budi Santoso", "20230001", 3.75);
  mhs1.tampilkanData();
}

Contoh Kode 5: Class Person dengan Properti Tersembunyi

class Person {
  String _name;
  int age;

  Person(this._name, this.age);

  void setName(String name) => _name = name;

  String getName() => _name;
}

void main() {
  Person person = Person("Salma", 20);

  print(person.age);
  person.setName("Aqila");
  print(person.getName());
}

Contoh Kode 6: Class Product dengan Konstruktor Sederhana

class Product {
  String nama;

  Product(this.nama);
}

void main() {
  Product produk1 = Product("Laptop Gaming");
  print(produk1.nama);
}

Contoh Kode 7: Class Produk dengan Konstruktor Bernama

class Produk {
  final String id, nama;
  final double harga;
  final int stok;
  final String kategori;

  Produk({
    required this.id,
    required this.nama,
    required this.harga,
    this.stok = 0,
    this.kategori = "Umum",
  });
}

void main() {
  Produk laptop = Produk(
    id: "P001",
    nama: "Laptop Gaming",
    harga: 15000000,
    stok: 15,
    kategori: "Elektronik",
  );

  Produk smartphone = Produk(
    id: "P002",
    nama: "Smartphone Flagship",
    harga: 12000000,
    stok: 25,
  );

  Produk kamera = Produk(
    id: "P003",
    nama: "Kamera Mirrorless",
    harga: 20000000,
  );

  print(laptop.nama);
  print(smartphone.nama);
  print(kamera.nama);
}
